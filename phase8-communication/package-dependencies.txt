# Add these dependencies to your package.json

# Install with npm:
npm install pusher pusher-js

# Or with yarn:
yarn add pusher pusher-js

# These packages provide:
# - pusher: Server-side library for triggering real-time events
# - pusher-js: Client-side library for receiving real-time events

# "pusher": "^5.1.3",
# "pusher-js": "^8.4.0-rc2"
